;; Some iterators for extending subst as much as possible
;; All vectors (Use it for destination)
(define_mode_iterator SUBST_V
  [V64QI V32QI V16QI
   V32HI V16HI V8HI
   V16SI V8SI  V4SI
   V8DI  V4DI  V2DI
   V32HF V16HF V8HF
   V32BF V16BF V8BF
   V16SF V8SF  V4SF
   V8DF  V4DF  V2DF])

(define_mode_iterator SUBST_CV
  [V32HF V16HF V8HF])

(define_mode_iterator SUBST_S
  [QI HI SI DI])

   V8DI  V4DI  V2DI
   V32HF V16HF V8HF
   V16SF V8SF  V4SF
   V8DF  V4DF  V2DF
   QI HI SI DI SF DF])
